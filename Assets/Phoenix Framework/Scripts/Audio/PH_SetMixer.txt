-- Register the behaviour
behaviour("PH_SetMixer")

function PH_SetMixer:Start()
	-- Run when behaviour is created
	local audioMixer = self.gameObject.GetComponent(DataContainer).GetString("audioMixer")
	self.gameObject.GetComponent(AudioSource).SetOutputAudioMixer(AudioSourceMixers[audioMixer])

end

AudioSourceMixers = {
	["master"] = AudioMixer.Music,
	["ingame"] = AudioMixer.Music,
	["important"] = AudioMixer.Music,
	["firstperson"] = AudioMixer.Music,
	["playervehicle"] = AudioMixer.Music,
	["world"] = AudioMixer.Music,
	["music"] = AudioMixer.Music,
	["musicsting"] = AudioMixer.Music,
}
